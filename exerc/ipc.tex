\def\IPC{
  \question[0,5] Dentre os principais mecanismos de {\bf comunicação}
  entre os processos (IPC -- {\em Interprocess communication}) listados
  na coluna da esquerda, associe à sua função descrita na coluna da
  direita.
  \bigskip
  
  \begin{minipage}{.25\textwidth}
    \begin{enumerate}
    \item {\em Pipes}
    \item {Mensagens}
    \item {Sinais}
    \end{enumerate}
  \end{minipage}
  \begin{minipage}{.75\textwidth}
    \small
    \begin{enumerate}[( )]
    \item Comunicação síncrona entre os processos utilizando descritores de arquivos.
    \item Comunicação assíncrona através de um sistema comum de fila.
    \item Comunicação direta e síncrona através do envio de mensagem ao processo.
    \end{enumerate}
  \end{minipage}
}
